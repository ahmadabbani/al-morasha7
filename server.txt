import express from "express";
import dotenv from "dotenv";
import getUsers from "./api/users/getUsers.js";

import createTable from "./api/users/createTable.js";
dotenv.config();

const app = express();
app.use(express.json());

app.use("/api/users/getUsers", getUsers);

// Mount API endpoints
app.use("/api/users/createTable", createTable);

// Serve React app (optional)
app.use(express.static("dist"));

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
